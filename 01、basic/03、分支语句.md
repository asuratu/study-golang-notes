```
const filename = "/Users/asura/Code/go/taobao/01、basic/abc.txt"
if contents, err := ioutil.ReadFile(filename); err != nil {

## 2. switch 条件语句
```
func eval(a, b int, op string) int {
	var result int
	switch op {
	case "+":
		result = a + b
	case "-":
		result = a - b
	case "*":
		result = a * b
	case "/":
		if b == 0 {
			panic("when a/b, b can`t 0")
		}
		result = a / b
	default:
		panic("unknown op")
	}
	return result
}
